Creating Functions:¶

  • Using bootstrapping we would be able to replicate existing values
  • This replications would help to generate normal distributions (Central Limit Theorem) and confindence intervals
In [59]:
def bootstrap_sample(data,func):
    
    bs_sample = np.random.choice(data,len(data))
    return func(bs_sample)
In [60]:
def replication(data):
    
    bs_replicate = np.empty(5000)

    for i in range(5000):
        bs_replicate[i] = bootstrap_sample(data, np.mean)
    
    return (
            np.percentile(bs_replicate,2.5), # 0
            np.round(np.mean(bs_replicate)), #1,
            np.percentile(bs_replicate,97.5), #2
            bs_replicate #3
            )
